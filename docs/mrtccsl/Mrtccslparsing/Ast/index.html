Module Mrtccslparsing.Ast

type id = string Loc.t
type var = string list Loc.t
type clock =
  1. | Clock
type percentage =
  1. | Percent of Number.Rational.t
type duration =
  1. | Second of {
    1. value : Number.Rational.t;
    2. scale : Number.Rational.t;
    }
type 'n distribution =
  1. | DUniform
  2. | DNormal of {
    1. mean : 'n;
    2. deviation : 'n;
    }
  3. | DExponential of {
    1. mean : 'n;
    }
type 'a interval = {
  1. left_strict : bool;
  2. left : 'a;
  3. right : 'a;
  4. right_strict : bool;
}
type 'a numeric_interval =
  1. | Interval of 'a interval
  2. | PlusMinus of 'a * 'a
type 'a inline_relation' =
  1. | InlineExpr of 'a
  2. | InlineInterval of 'a numeric_interval
type 'a inline_relation = 'a inline_relation' Loc.t
type additive_union =
  1. | AUnion
  2. | ADisjunctiveUnion
val compare_additive_union : additive_union -> additive_union -> int
type clock_rel =
  1. | Coincidence
  2. | Exclusion
  3. | Precedence
  4. | Causality
  5. | Subclocking of percentage option
  6. | Alternation of bool
and contdist = duration distribution Loc.t
and duration_expr' =
  1. | DConstant of duration
  2. | DVariable of var
  3. | DBinOp of duration_expr * [ `Add | `Sub ] * duration_expr
  4. | DScaleOp of Number.Rational.t * duration_expr
  5. | DNegation of duration_expr
  6. | DHole
and duration_expr = duration_expr' Loc.t
and int_expr' =
  1. | IConstant of int
  2. | IVariable of var
  3. | IBinOp of int_expr * Expr.num_op * int_expr
  4. | INegation of int_expr
  5. | IHole
and int_expr = int_expr' Loc.t
and clock_expr' =
  1. | CVariable of var
  2. | CIntersection of clock_expr list
  3. | CUnion of clock_expr list
  4. | CDisjUnion of {
    1. args : clock_expr list;
    2. ratios : var option;
    }
  5. | CTickDelay of {
    1. arg : clock_expr;
    2. delay : int_expr inline_relation;
    3. on : clock_expr option;
    }
  6. | CNext of clock_expr
  7. | CPeriodic of {
    1. base : clock_expr;
    2. period : int Loc.t;
    3. error : int_expr inline_relation;
    4. offset : int_expr inline_relation option;
    }
  8. | CSample of {
    1. arg : clock_expr;
    2. base : clock_expr;
    }
  9. | CFirstSample of {
    1. arg : clock_expr;
    2. base : clock_expr;
    }
  10. | CLastSample of {
    1. arg : clock_expr;
    2. base : clock_expr;
    }
  11. | CMinus of {
    1. base : clock_expr;
    2. subs : clock_expr list;
    }
  12. | CFastest of clock_expr list
  13. | CSlowest of clock_expr list
  14. | CPeriodJitter of {
    1. period : duration;
    2. error : duration_expr inline_relation;
    3. offset : duration_expr option;
    }
  15. | CPeriodDrift of {
    1. period : duration;
    2. error : duration_expr inline_relation;
    3. offset : duration_expr option;
    }
  16. | CTimeDelay of {
    1. arg : clock_expr;
    2. delay : duration_expr inline_relation;
    }
  17. | CSporadic of {
    1. at_least : duration;
    2. strict : bool;
    }
and clock_expr = clock_expr' Loc.t
type var_type = [
  1. | `Int
  2. | `Duration
  3. | `Clock
]
type statement' =
  1. | VariableDeclaration of (id list * var_type) list
  2. | IntRelation of int_expr * (Expr.num_rel * int_expr) list
  3. | DurationRelation of duration_expr * (Expr.num_rel * duration_expr) list
  4. | ClockRelation of clock_expr * clock_rel * clock_expr
  5. | AdditiveUnion of var * additive_union * clock_expr
  6. | DiscreteValued of {
    1. var : var;
    2. values : int list Loc.t;
    3. ratios : int list Loc.t;
    }
  7. | ContinuousValued of {
    1. var : var;
    2. dist : contdist;
    }
  8. | Pool of int * (clock_expr * clock_expr) list
  9. | Block of {
    1. name : id;
    2. statements : statements;
    }
  10. | Allow of {
    1. clocks : clock_expr list;
    2. left : clock_expr;
    3. right : clock_expr;
    }
  11. | Forbid of {
    1. clocks : clock_expr list;
    2. left : clock_expr;
    3. right : clock_expr;
    }
and statement = statement' Loc.t
and statements = statement list
type module_body_declaration = {
  1. assumptions : statements list;
  2. structure : statements;
  3. assertions : statements list;
}
type id_types = [
  1. | `Duration
  2. | `Int
  3. | `Block
  4. | `Clock
]
