type ('c, 'tp, 'ip, 'tv, 'iv, 't) clock_constr =
  1. | Precedence of {
    1. cause : 'c;
    2. conseq : 'c;
    }
  2. | Causality of {
    1. cause : 'c;
    2. conseq : 'c;
    }
  3. | Exclusion of {
    1. args : 'c list;
    2. choice : 'iv option;
    }
  4. | Coincidence of 'c list
  5. | Subclocking of {
    1. sub : 'c;
    2. super : 'c;
    3. choice : 'iv option;
    }
  6. | Minus of {
    1. out : 'c;
    2. arg : 'c;
    3. except : 'c list;
    }
  7. | Delay of {
    1. out : 'c;
    2. arg : 'c;
    3. delay : ('iv, int) arg;
    4. base : 'c;
    }
  8. | Fastest of {
    1. out : 'c;
    2. args : 'c list;
    }
  9. | Slowest of {
    1. out : 'c;
    2. args : 'c list;
    }
  10. | Intersection of {
    1. out : 'c;
    2. args : 'c list;
    }
  11. | Union of {
    1. out : 'c;
    2. args : 'c list;
    }
  12. | Periodic of {
    1. out : 'c;
    2. base : 'c;
    3. period : int;
    4. error : ('iv, int) arg;
    5. offset : ('ip, int) arg;
    }
  13. | Sample of {
    1. out : 'c;
    2. arg : 'c;
    3. base : 'c;
    }
  14. | Alternate of {
    1. first : 'c;
    2. second : 'c;
    3. strict : bool;
    }
  15. | RTdelay of {
    1. out : 'c;
    2. arg : 'c;
    3. delay : ('tv, 't) arg;
    }
  16. | CumulPeriodic of {
    1. out : 'c;
    2. period : 't;
    3. error : ('tv, 't) arg;
    4. offset : ('tp, 't) arg;
    }
  17. | AbsPeriodic of {
    1. out : 'c;
    2. period : 't;
    3. error : ('tv, 't) arg;
    4. offset : ('tp, 't) arg;
    }
  18. | FirstSampled of {
    1. out : 'c;
    2. arg : 'c;
    3. base : 'c;
    }
  19. | LastSampled of {
    1. out : 'c;
    2. arg : 'c;
    3. base : 'c;
    }
  20. | Forbid of {
    1. left : 'c;
    2. right : 'c;
    3. args : 'c list;
    }
  21. | Allow of {
    1. left : 'c;
    2. right : 'c;
    3. args : 'c list;
    }
  22. | Sporadic of {
    1. out : 'c;
    2. at_least : ('tp, 't) arg;
    3. strict : bool;
    }
  23. | Pool of int * ('c * 'c) list
    (*

    Mutex is a special case of Pool where n=1

    *)
  24. | DisjunctiveUnion of {
    1. out : 'c;
    2. args : 'c list;
    3. choice : 'iv option;
    }

Clock constraint type. 'c is clock variable symbols type, '_p is parameter (unchanged during execution) symbols type ('tp is for durations, 'ip is for integers), '_v is symbol type for variables (change during execution), 't is the type of duration constants.
